    """
    Main loop that runs on the RPi 5.
    """
    while True:
        # 1. Update the Idle Gaze logic
        self.gaze_controller.update_idle_gaze()

        # This draws the layers defined in your architecture
        self.render_all_layers()
        
        time.sleep(0.01) # Maintain 60fps
